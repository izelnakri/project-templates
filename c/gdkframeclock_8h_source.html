46/**
47 * GdkFrameClockPhase:
48 * @GDK_FRAME_CLOCK_PHASE_NONE: no phase
49 * @GDK_FRAME_CLOCK_PHASE_FLUSH_EVENTS: corresponds to GdkFrameClock::flush-events. Should not be handled by applications.
50 * @GDK_FRAME_CLOCK_PHASE_BEFORE_PAINT: corresponds to GdkFrameClock::before-paint. Should not be handled by applications.
51 * @GDK_FRAME_CLOCK_PHASE_UPDATE: corresponds to GdkFrameClock::update.
52 * @GDK_FRAME_CLOCK_PHASE_LAYOUT: corresponds to GdkFrameClock::layout. Should not be handled by applications.
53 * @GDK_FRAME_CLOCK_PHASE_PAINT: corresponds to GdkFrameClock::paint.
54 * @GDK_FRAME_CLOCK_PHASE_RESUME_EVENTS: corresponds to GdkFrameClock::resume-events. Should not be handled by applications.
55 * @GDK_FRAME_CLOCK_PHASE_AFTER_PAINT: corresponds to GdkFrameClock::after-paint. Should not be handled by applications.
56 *
57 * Used to represent the different paint clock phases that can be requested.
58 *
59 * The elements of the enumeration correspond to the signals of `GdkFrameClock`.
60 */
61typedef enum {
62 GDK_FRAME_CLOCK_PHASE_NONE = 0,
63 GDK_FRAME_CLOCK_PHASE_FLUSH_EVENTS = 1 << 0,
64 GDK_FRAME_CLOCK_PHASE_BEFORE_PAINT = 1 << 1,
65 GDK_FRAME_CLOCK_PHASE_UPDATE = 1 << 2,
66 GDK_FRAME_CLOCK_PHASE_LAYOUT = 1 << 3,
67 GDK_FRAME_CLOCK_PHASE_PAINT = 1 << 4,
68 GDK_FRAME_CLOCK_PHASE_RESUME_EVENTS = 1 << 5,
69 GDK_FRAME_CLOCK_PHASE_AFTER_PAINT = 1 << 6
70} GdkFrameClockPhase;
