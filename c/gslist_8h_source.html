145#define g_clear_slist(slist_ptr, destroy) \
146 G_STMT_START { \
147 GSList *_slist; \
148 \
149 _slist = *(slist_ptr); \
150 if (_slist) \
151 { \
152 *slist_ptr = NULL; \
153 \
154 if ((destroy) != NULL) \
155 g_slist_free_full (_slist, (destroy)); \
156 else \
157 g_slist_free (_slist); \
158 } \
159 } G_STMT_END \
