290static inline char *CRI_USER_TAG_ID(tostr, iptr)(intptr_t *e)
291{
292 uintptr_t absptr = (uintptr_t) *e;
293 if (absptr > (uintptr_t)INTPTR_MAX)
294 absptr = -absptr;
295
296 char *str = NULL;
297 cr_asprintf(&str, "%s0x%" CRI_PRIxPTR, *e < 0 ? "-" : "", absptr);
298 return str;
